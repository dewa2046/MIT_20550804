CREATE TABLE BOCATM.ATM_INFO
(
  ATM_NAME        VARCHAR2(10 BYTE)             NOT NULL,
  BR_CODE         INTEGER,
  LOCATION        VARCHAR2(100 BYTE),
  IP              VARCHAR2(30 BYTE),
  OS              VARCHAR2(20 BYTE),
  MODEL           INTEGER,
  VENDER          INTEGER,
  INST_DATE       DATE,
  DIV_TYPE        VARCHAR2(10 BYTE),
  SERIAL          VARCHAR2(20 BYTE),
  EMV             INTEGER,
  RECYCLER        INTEGER,
  STATUS          INTEGER,
  REMOTE          INTEGER,
  OVERSEAS        INTEGER,
  PROTOCOL        INTEGER,
  LAST_UPDATE     DATE,
  LAST_UPDATE_BY  INTEGER
)

PCTUSED    0
PCTFREE    10
INITRANS   1
MAXTRANS   255

            INITIAL          64K
            NEXT             1M
            MINEXTENTS       1
            MAXEXTENTS       UNLIMITED
            PCTINCREASE      0
            BUFFER_POOL      DEFAULT
           )
LOGGING 
NOCOMPRESS 
NOCACHE
NOPARALLEL
MONITORING;

COMMENT ON COLUMN BOCATM.ATM_INFO.STATUS IS '1-ACTIVE,0-INACTIVE';

COMMENT ON COLUMN BOCATM.ATM_INFO.REMOTE IS '0-NO,1-YES';

COMMENT ON COLUMN BOCATM.ATM_INFO.OVERSEAS IS '0-NO,1-YES';
